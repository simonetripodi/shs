77      @Test
78      public void verifyParsedHeaders()
79          throws Exception
80      {
81          String simpleRequest = "GET /index.html HTTP/1.1\n"
82                                  + "Host: www.google.nl\n"
83                                  + "Accept: text/xml,application/xml,application/xhtml+xml,text/html;q=0.9,text/plain;q=0.8,image/png,*/*;q=0.5\n"
84                                  + "Accept-Language: en-us,en;q=0.5\n"
85                                  + "Accept-Encoding: gzip,deflate\n"
86                                  + "Accept-Charset: ISO-8859-1,utf-8;q=0.7,*;q=0.7\n"
87                                  + "Keep-Alive: 300\n"
88                                  + "Connection: keep-alive\n"
89                                  + "Referer: http://www.google.nl/index.html\n\n";
90          Request request = parse( simpleRequest );
91  
92          assertEquals( "www.google.nl", request.getHeaders().getFirstValue( "Host" ) );
93  
94          // Accept header
95          assertTrue( request.getHeaders().getValues( ACCEPT ).contains( "text/xml" ) );
96          assertTrue( request.getHeaders().getValues( ACCEPT ).contains( "application/xml" ) );
97          assertTrue( request.getHeaders().getValues( ACCEPT ).contains( "application/xhtml+xml" ) );
98          assertTrue( request.getHeaders().getValues( ACCEPT ).contains( "text/html;q=0.9" ) );
99          assertTrue( request.getHeaders().getValues( ACCEPT ).contains( "text/plain;q=0.8" ) );
100         assertTrue( request.getHeaders().getValues( ACCEPT ).contains( "image/png" ) );
101         assertTrue( request.getHeaders().getValues( ACCEPT ).contains( "*/*;q=0.5" ) );
102 
103         // Accept-Language
104         assertTrue( request.getHeaders().getValues( ACCEPT_LANGUAGE ).contains( "en-us" ) );
105         assertTrue( request.getHeaders().getValues( ACCEPT_LANGUAGE ).contains( "en;q=0.5" ) );
106 
107         // Accept-Encoding
108         assertTrue( request.getHeaders().getValues( ACCEPT_ENCODING ).contains( "gzip" ) );
109         assertTrue( request.getHeaders().getValues( ACCEPT_ENCODING ).contains( "deflate" ) );
110 
111         // Accept-Charset
112         assertTrue( request.getHeaders().getValues( ACCEPT_CHARSET ).contains( "ISO-8859-1" ) );
113         assertTrue( request.getHeaders().getValues( ACCEPT_CHARSET ).contains( "utf-8;q=0.7" ) );
114         assertTrue( request.getHeaders().getValues( ACCEPT_CHARSET ).contains( "*;q=0.7" ) );
115 
116         assertEquals( "300", request.getHeaders().getFirstValue( KEEP_ALIVE ) );
117         assertEquals( "keep-alive", request.getHeaders().getFirstValue( CONNECTION ) );
118         assertEquals( "http://www.google.nl/index.html", request.getHeaders().getFirstValue( REFERER ) );
119     }

180     private Request parse( String mockRequestString )
181         throws Exception
182     {
183         RequestStreamingParser pullParser = new RequestStreamingParser( "localhost", "localhost", 123 );
184 
185         for ( String chunk : mockRequestString.split("(?<=\\G.{4})") )
186         {
187             pullParser.onRequestPartRead( utf8ByteBuffer( chunk ) );
188         }
189 
190         assertTrue( pullParser.isRequestMessageComplete() );
191 
192         return pullParser.getParsedRequest();
193     }
