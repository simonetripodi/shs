53  /**
54   * Asynchronous HTTP protocol processor to serve current request.
55   */
56  final class ProtocolProcessor
57      implements Runnable
58  {
59  
60      private static final Logger logger = getLogger( ProtocolProcessor.class );
61  
62      private static final String DEFAULT_SERVER_NAME = "Simple HttpServer";
63  
64      private static final String GZIP = "gzip";
65  
66      private static final String HTTP_11 = "1.1";
67  
68      private static final String KEEP_ALIVE = "Keep-Alive";
69  
70      private final SimpleDateFormat dateFormat = new SimpleDateFormat( "EEE, dd MMM yyyy HH:mm:ss zzz" );
71  
72      private final SessionManager sessionManager;
73  
74      private final RequestDispatcher requestDispatcher;
75  
76      private final Request request;
77  
78      private SelectionKey key;
79  
80      public ProtocolProcessor( SessionManager sessionManager,
81                                RequestDispatcher requestDispatcher,
82                                Request request,
83                                SelectionKey key )
84      {
85          this.sessionManager = sessionManager;
86          this.requestDispatcher = requestDispatcher;
87          this.request = request;
88          this.key = key;
89      }
90  
91      public void run()
92      {
93          long start = currentTimeMillis();
94  
95          // debug the request
96          if ( logger.isDebugEnabled() )
97          {
98              logger.debug( "< {} {} {}/{}", new Object[] {
99                                                 request.getMethod(),
100                                                request.getPath(),
101                                                request.getProtocolName(),
102                                                request.getProtocolVersion() } );
103 
104             for ( Entry<String, List<String>> header : request.getHeaders().getAllEntries() )
105             {
106                 Formatter headerValues = new Formatter();
107                 int i = 0;
108                 for ( String value : header.getValue() )
109                 {
110                     headerValues.format( "%s%s", ( i++ > 0 ? ", " : "" ), value );
111                 }
112 
113                 logger.debug( "< {}: {}", header.getKey(), headerValues.toString() );
114             }
115 
116             if ( !request.getCookies().isEmpty() )
117             {
118                 Formatter cookies = new Formatter();
119                 int counter = 0;
120                 for ( Cookie cookie : request.getCookies() )
121                 {
122                     cookies.format( "%s%s=%s", ( counter++ > 0 ? ", " : "" ), cookie.getName(), cookie.getValue() );
123                 }
124                 logger.debug( "< Cookie: {}", cookies.toString() );
125             }
126         }
127 
128         Response response = newResponse();
129         response.addHeader( DATE, dateFormat.format( new Date() ) );
130         response.addHeader( SERVER, DEFAULT_SERVER_NAME );
131 
132         boolean keepAlive = HTTP_11.equals( request.getProtocolVersion() )
133                             || ( request.getHeaders().contains( CONNECTION )
134                                  && KEEP_ALIVE.equals( request.getHeaders().getFirstValue( KEEP_ALIVE ) ) );
135         if ( keepAlive )
136         {
137             response.addHeader( CONNECTION, KEEP_ALIVE );
138             // response.addHeader( KEEP_ALIVE, format( "timeout=%s", socket.getSoTimeout() / 1000 ) );
139         }
140         boolean gzipEnabled = request.getHeaders().contains( ACCEPT_ENCODING )
141                               && request.getHeaders().getValues( ACCEPT_ENCODING ).contains( GZIP );
142 
143         try
144         {
145             sessionManager.manageSession( request, response );
146             requestDispatcher.dispatch( request, response );
147 
148             response.setProtocolName( request.getProtocolName() );
149             response.setProtocolVersion( request.getProtocolVersion() );
150 
151             if ( response.getBodyWriter().getContentLength() > 0 )
152             {
153                 response.addHeader( CONTENT_LENGTH, String.valueOf( response.getBodyWriter().getContentLength() ) );
154 
155                 if ( response.getBodyWriter().contentType() != null )
156                 {
157                     response.addHeader( CONTENT_TYPE, response.getBodyWriter().contentType() );
158                 }
159 
160                 if ( gzipEnabled )
161                 {
162                     response.addHeader( CONTENT_ENCODING, GZIP );
163                 }
164             }
165         }
166         catch ( IOException e )
167         {
168             logger.error( "Request cannot be satisfied due to internal I/O error", e );
169 
170             response.setStatus( INTERNAL_SERVER_ERROR );
171         }
172         finally
173         {
174             long time = currentTimeMillis() - start;
175 
176             // debug the response
177             if ( logger.isDebugEnabled() )
178             {
179                 // protocol
180                 logger.debug( "> {}/{} {} {}",
181                               new Object[] {
182                                   response.getProtocolName(),
183                                   response.getProtocolVersion(),
184                                   response.getStatus().getStatusCode(),
185                                   response.getStatus().getStatusText() } );
186                 // headers
187                 for ( Entry<String, List<String>> header : response.getHeaders().getAllEntries() )
188                 {
189                     Formatter headerValues = new Formatter();
190 
191                     int counter = 0;
192                     for ( String headerValue : header.getValue() )
193                     {
194                         headerValues.format( "%s%s", (counter++ > 0 ? ", " : ""), headerValue );
195                     }
196 
197                     logger.debug( "> {}: {}", header.getKey(), headerValues.toString() );
198                 }
199                 // cookies
200                 for ( Cookie cookie : response.getCookies() )
201                 {
202                     Formatter cookieFormatter = new Formatter()
203                                              .format( "%s=%s; Path=%s; Domain=%s;",
204                                                       cookie.getName(), cookie.getValue(), cookie.getPath(), cookie.getDomain() );
205 
206                     if ( !cookie.getPorts().isEmpty() )
207                     {
208                         cookieFormatter.format( " Port=\"" );
209                         int i = 0;
210                         for ( Integer port : cookie.getPorts() )
211                         {
212                             cookieFormatter.format( "%s%s", ( i++ > 0 ? "," : "" ), port );
213                         }
214                         cookieFormatter.format( "\";" );
215                     }
216 
217                     if ( cookie.getMaxAge() != -1 )
218                     {
219                         Date expirationDate = new Date( cookie.getMaxAge() * 1000 + currentTimeMillis() );
220                         String expires = dateFormat.format( expirationDate );
221 
222                         cookieFormatter.format( " Expires=%s;", expires );
223                     }
224 
225                     // secure field ignored since HTTPs is not supported in this version
226 
227                     logger.debug( "> Set-Cookie: {} HttpOnly", cookieFormatter.toString() );
228 
229                     logger.debug( "Request processed in {}ms", time );
230                 }
231             }
232 
233             try
234             {
235                 new ResponseSerializer( key, gzipEnabled ).serialize( response );
236             }
237             catch ( IOException e )
238             {
239                 key.cancel();
240 
241                 logger.error( "Impossible to stream Response to the client", e );
242             }
243         }
244     }
