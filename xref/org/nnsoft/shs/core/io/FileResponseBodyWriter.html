217     /**
218      * The file has to be transferred to the body response.
219      */
220     private final File toBeTransfered;
221 
222     /**
223      * Creates a new ResponseBodyWriter given the file has to be transferred to the body response.
224      *
225      * @param toBeTransfered the file has to be transferred to the body response.
226      */
227     public FileResponseBodyWriter( File toBeTransfered )
228     {
229         this.toBeTransfered = toBeTransfered;
230     }
231 
232     /**
233      * {@inheritDoc}
234      */
235     public long getContentLength()
236     {
237         return toBeTransfered.length();
238     }
239 
240     /**
241      * {@inheritDoc}
242      */
243     @Override
244     public String contentType()
245     {
246         int extensionSeparator = toBeTransfered.getName().lastIndexOf( '.' );
247         String extension = toBeTransfered.getName().substring( ++extensionSeparator );
248         return CONTENT_TYPES.get( extension );
249     }
250 
251     /**
252      * {@inheritDoc}
253      */
254     public void write( WritableByteChannel target )
255         throws IOException
256     {
257         FileChannel channel = new FileInputStream( toBeTransfered ).getChannel();
258         try
259         {
260             channel.transferTo( 0, toBeTransfered.length(), target );
261         }
262         finally
263         {
264             closeQuietly( channel );
265         }
266     }
