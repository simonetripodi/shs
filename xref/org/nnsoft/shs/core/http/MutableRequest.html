330     /**
331      * {@inheritDoc}
332      */
333     public <T> T readRequestBody( RequestBodyReader<T> requestBodyReader )
334         throws IOException
335     {
336         checkArgument( requestBodyReader != null, "Null requestBodyReader not allowed" );
337 
338         if ( requestBody == null )
339         {
340             throw new StreamAlreadyConsumedException();
341         }
342 
343         requestBody.rewind();
344 
345         return requestBodyReader.onBodyPartReceived( requestBody );
346     }
